CREATE TABLE Blodgiver (
    blodgiver_id int NOT NULL AUTO_INCREMENT,
    fornavn varchar(30),
    etternavn varchar(30),
    fodseldato date,
	adresse varchar(30),
    post_nr  int(4),
    post_sted  varchar(30),
	personnummer int (5),
	telefon int(30),
	epost varchar(30),
	forrige_blodtapp date,
	godkjent_egenerklering TINYINT,
	karantene int(30),
	passord varchar(30),
	blodtype varchar(3),
	PRIMARY KEY (blodgiver_id)
)

)

CREATE TABLE Blodegenskap (
